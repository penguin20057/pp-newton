\documentclass[12pt]{article} \usepackage[a4paper, margin=1in]{geometry} \usepackage[vietnamese]{babel} \usepackage{amsmath} \usepackage{amsfonts} \usepackage{enumitem} \usepackage[utf8]{vietnam} \begin{document} \begin{center} \textbf{ĐỀ THI TRẮC NGHIỆM: PHƯƠNG TRÌNH TIẾP TUYẾN VÀ PHƯƠNG PHÁP NEWTON} \end{center} \begin{enumerate}[label=\textbf{Câu \arabic*:}] % Câu 1: Phương trình tiếp tuyến cơ bản \item Phương trình tiếp tuyến của đồ thị hàm số $y = x^2 + 2x + 1$ tại điểm có hoành độ $x = 1$ là: \begin{enumerate}[label=\Alph*.] \item $y = 4x - 2$ \item $y = 4x - 1$ \item $y = 2x + 1$ \item $y = 2x - 1$ \end{enumerate} % Câu 2: Tính đạo hàm để tìm phương trình tiếp tuyến \item Cho hàm số $y = \sin(x)$ tại điểm $x = \frac{\pi}{2}$. Phương trình tiếp tuyến tại điểm này là: \begin{enumerate}[label=\Alph*.] \item $y = x - \frac{\pi}{2}$ \item $y = 1$ \item $y = -x + \frac{\pi}{2} + 1$ \item $y = x + 1$ \end{enumerate} % Câu 3: Tiếp tuyến song song với đường thẳng \item Tiếp tuyến của đồ thị hàm số $y = x^3 - 3x + 2$ song song với đường thẳng $y = 9x + 5$. Giá trị $x$ tại điểm tiếp xúc là: \begin{enumerate}[label=\Alph*.] \item $x = 1$ \item $x = -1$ \item $x = 2$ \item $x = -2$ \end{enumerate} % Câu 4: Tiếp tuyến vuông góc với đường thẳng \item Điểm tiếp xúc của tiếp tuyến vuông góc với đường thẳng $y = -2x + 3$ trên đồ thị $y = x^2$ có hoành độ là: \begin{enumerate}[label=\Alph*.] \item $x = \frac{1}{4}$ \item $x = -\frac{1}{4}$ \item $x = \frac{1}{2}$ \item $x = -\frac{1}{2}$ \end{enumerate} % Câu 5: Phương trình tiếp tuyến tại điểm cho trước \item Phương trình tiếp tuyến của đồ thị $y = e^x$ tại điểm có tung độ $y = e$ là: \begin{enumerate}[label=\Alph*.] \item $y = ex$ \item $y = x + e$ \item $y = ex + e$ \item $y = e(x - 1) + e$ \end{enumerate} % Câu 6: Phương pháp Newton cơ bản \item Sử dụng phương pháp Newton để tìm nghiệm gần đúng của phương trình $x^3 - x - 2 = 0$ với giá trị ban đầu $x_0 = 1$. Giá trị $x_1$ là: \begin{enumerate}[label=\Alph*.] \item $x_1 = 1.5$ \item $x_1 = 1.25$ \item $x_1 = 2$ \item $x_1 = 1.75$ \end{enumerate} % Câu 7: Công thức lặp Newton \item Công thức lặp của phương pháp Newton cho phương trình $f(x) = x^2 - 4$ là: \begin{enumerate}[label=\Alph*.] \item $x_{n+1} = x_n - \frac{x_n^2 - 4}{2x_n}$ \item $x_{n+1} = x_n - \frac{2x_n}{x_n^2 - 4}$ \item $x_{n+1} = x_n + \frac{x_n^2 - 4}{2x_n}$ \item $x_{n+1} = x_n - \frac{x_n^2 + 4}{2x_n}$ \end{enumerate} % Câu 8: Ứng dụng phương pháp Newton \item Tìm nghiệm gần đúng của phương trình $\cos(x) = x$ với giá trị ban đầu $x_0 = 0.5$. Giá trị $x_1$ theo phương pháp Newton là (làm tròn đến 3 chữ số thập phân): \begin{enumerate}[label=\Alph*.] \item $x_1 = 0.755$ \item $x_1 = 0.739$ \item $x_1 = 0.725$ \item $x_1 = 0.750$ \end{enumerate} % Câu 9: Kết hợp tiếp tuyến và Newton \item Cho hàm $f(x) = x^3 - 3x + 1$. Đường tiếp tuyến tại $x = 1$ cắt trục hoành tại điểm có hoành độ $x_1$. Giá trị $x_1$ này có thể được dùng làm giá trị ban đầu cho phương pháp Newton. Giá trị $x_1$ là: \begin{enumerate}[label=\Alph*.] \item $x_1 = 0$ \item $x_1 = 1$ \item $x_1 = \frac{1}{3}$ \item $x_1 = -\frac{1}{3}$ \end{enumerate} % Câu 10: Ứng dụng nâng cao phương pháp Newton \item Khi sử dụng phương pháp Newton để tìm nghiệm của $f(x) = x^4 - 4x^2 + 4$ với $x_0 = 1$, giá trị $x_2$ (làm tròn đến 3 chữ số thập phân) là: \begin{enumerate}[label=\Alph*.] \item $x_2 = 1.414$ \item $x_2 = 1.500$ \item $x_2 = 1.732$ \item $x_2 = 2.000$ \end{enumerate} \end{enumerate} \textbf{Hướng dẫn làm bài:} - Chọn đáp án đúng nhất cho mỗi câu hỏi. - Mỗi câu trả lời đúng được 1 điểm. - Thời gian làm bài: 45 phút. \end{document}
